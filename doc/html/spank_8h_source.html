49 /* SPANK plugin operations. SPANK plugin should have at least one of
50  * these functions defined non-NULL.
51  *
52  * Plug-in callbacks are completed at the following points in slurmd:
53  *
54  * slurmd
55  * `-> slurmd_init()
56  * |
57  * `-> job_prolog()
58  * |
59  * | `-> slurmstepd
60  * | `-> init ()
61  * | -> process spank options
62  * | -> init_post_opt ()
63  * | + drop privileges (initgroups(), seteuid(), chdir())
64  * | `-> user_init ()
65  * | + for each task
66  * | | + fork ()
67  * | | |
68  * | | + reclaim privileges
69  * | | `-> task_init_privileged ()
70  * | | |
71  * | | + become_user ()
72  * | | `-> task_init ()
73  * | | |
74  * | | + execve ()
75  * | |
76  * | + reclaim privileges
77  * | + for each task
78  * | | `-> task_post_fork ()
79  * | |
80  * | + for each task
81  * | | + wait ()
82  * | | `-> task_exit ()
83  * | `-> exit ()
84  * |
85  * `---> job_epilog()
86  * |
87  * `-> slurmd_exit()
88  *
89  * In srun only the init(), init_post_opt() and local_user_init(), and exit()
90  * callbacks are used.
91  *
92  * In sbatch/salloc only the init(), init_post_opt(), and exit() callbacks
93  * are used.
94  *
95  * In slurmd proper, only the slurmd_init(), slurmd_exit(), and
96  * job_prolog/epilog callbacks are used.
97  *
98  */

115 /* Items which may be obtained from the spank handle using the
116  * spank_get_item () call. The expected list of variable arguments may
117  * be found in the comments below.
118  *
119  * For example, S_JOB_NCPUS takes (uint16_t *), a pointer to uint16_t, so
120  * the get item call would look like:
121  *
122  * uint16_t ncpus;
123  * spank_err_t rc = spank_get_item (spank, S_JOB_NCPUS, &ncpus);
124  *
125  * while S_JOB_PID_TO_GLOBAL_ID takes (pid_t, uint32_t *), so it would
126  * be called as:
127  *
128  * uint32_t global_id;
129  * spank_err_t rc;
130  * rc = spank_get_item (spank, S_JOB_PID_TO_GLOBAL_ID, pid, &global_id);
131  */
132 enum spank_item {
133  S_JOB_UID, /* User id (uid_t *) */
134  S_JOB_GID, /* Primary group id (gid_t *) */
135  S_JOB_ID, /* SLURM job id (uint32_t *) */
136  S_JOB_STEPID, /* SLURM job step id (uint32_t *) */
137  S_JOB_NNODES, /* Total number of nodes in job (uint32_t *) */
138  S_JOB_NODEID, /* Relative id of this node (uint32_t *) */
139  S_JOB_LOCAL_TASK_COUNT, /* Number of local tasks (uint32_t *) */
140  S_JOB_TOTAL_TASK_COUNT, /* Total number of tasks in job (uint32_t *) */
141  S_JOB_NCPUS, /* Number of CPUs used by this job (uint16_t *) */
142  S_JOB_ARGV, /* Command args (int *, char ***) */
143  S_JOB_ENV, /* Job env array (char ***) */
144  S_TASK_ID, /* Local task id (int *) */
145  S_TASK_GLOBAL_ID, /* Global task id (uint32_t *) */
146  S_TASK_EXIT_STATUS, /* Exit status of task if exited (int *) */
147  S_TASK_PID, /* Task pid (pid_t *) */
148  S_JOB_PID_TO_GLOBAL_ID, /* global task id from pid (pid_t, uint32_t *) */
149  S_JOB_PID_TO_LOCAL_ID, /* local task id from pid (pid_t, uint32_t *) */
150  S_JOB_LOCAL_TO_GLOBAL_ID,/* local id to global id (uint32_t, uint32_t *) */
151  S_JOB_GLOBAL_TO_LOCAL_ID,/* global id to local id (uint32_t, uint32_t *) */
152  S_JOB_SUPPLEMENTARY_GIDS,/* Array of suppl. gids (gid_t **, int *) */
153  S_SLURM_VERSION, /* Current SLURM version (char **) */
154  S_SLURM_VERSION_MAJOR, /* SLURM version major release (char **) */
155  S_SLURM_VERSION_MINOR, /* SLURM version minor release (char **) */
156  S_SLURM_VERSION_MICRO, /* SLURM version micro release (char **) */
157  S_STEP_CPUS_PER_TASK, /* CPUs allocated per task (=1 if --overcommit
158  * option is used, uint32_t *) */
159  S_JOB_ALLOC_CORES, /* Job allocated cores in list format (char **) */
160  S_JOB_ALLOC_MEM, /* Job allocated memory in MB (uint64_t *) */
161  S_STEP_ALLOC_CORES, /* Step alloc'd cores in list format (char **) */
162  S_STEP_ALLOC_MEM, /* Step alloc'd memory in MB (uint64_t *) */
163  S_SLURM_RESTART_COUNT /* Job restart count (uint32_t *) */
164 };
